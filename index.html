// バッジの色判定ロジック：AIが出力した文字列に "not ok" が含まれるかで判定 function getBadgeClass(val) { if (!val) return "badge-ok"; return val.toLowerCase().includes("not ok") ? "badge-not-ok" : "badge-ok"; } // 描画処理：AIが出力した主語(I'm/You're)付きの文字列をそのまま表示に反映 function renderResult(data) { const resultArea = document.getElementById('result-area'); resultArea.classList.remove('hidden'); document.getElementById('game-name').innerText = data.game_name || "不明なゲーム"; document.getElementById('definition').innerText = data.definition || "-"; document.getElementById('scenario').innerText = data.scenario || "-"; const sSelf = data.position_start?.self || "I'm OK"; const sOthers = data.position_start?.others || "You're OK"; const eSelf = data.position_end?.self || "I'm OK"; const eOthers = data.position_end?.others || "You're OK"; const posStartSelf = document.getElementById('pos-start-self'); const posStartOthers = document.getElementById('pos-start-others'); const posEndSelf = document.getElementById('pos-end-self'); const posEndOthers = document.getElementById('pos-end-others'); posStartSelf.innerText = sSelf; posStartSelf.className = `ok-badge ${getBadgeClass(sSelf)}`; posStartOthers.innerText = sOthers; posStartOthers.className = `ok-badge ${getBadgeClass(sOthers)}`; posEndSelf.innerText = eSelf; posEndSelf.className = `ok-badge ${getBadgeClass(eSelf)}`; posEndOthers.innerText = eOthers; posEndOthers.className = `ok-badge ${getBadgeClass(eOthers)}`; document.getElementById('prediction').innerText = data.prediction || "-"; document.getElementById('hidden-motive').innerText = data.hidden_motive || "-"; document.getElementById('advice').innerText = data.advice || "-"; document.getElementById('counseling-msg').innerText = data.counseling_msg || "-"; const sName = data.subject_name || "他者"; document.getElementById('label-start').innerHTML = `▶ 開始時 (対${sName})`; document.getElementById('label-end').innerHTML = `▼ 結末 (対${sName})`; }
